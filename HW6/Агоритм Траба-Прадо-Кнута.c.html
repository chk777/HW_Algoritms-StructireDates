#include <stdio.h>
#include <math.h>

/* f(x) = sqrt(|x|) + 5*x**3 */
double f (double x) {
    return (sqrt(fabs(x)) + 5.0*pow(x,3.0));
}

int main (int argc, char* argv[]) {
double A [11], y;
int i;

// Чтение в значениях массива A
for (i=0; i<11; i++) {
scanf ("%lf", &A[i]);
}

//В обратном порядке применяем "f" к каждому элементу A и выводим на печать 
    for (i=10; i>=0; i--) {
        y = f (A[i]);
    if (y > 400.0) {
        printf ("%d TOO LARGE\n", i);
                    } 
        else {
        printf ("%d %f\n", i, y);
             }
                           }
         return (0);
}
